# CadIR Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Insert an intermediate representation (IR) layer between the Python parametric design code and the build123d/OCCT backend, enabling caching, swappable backends, and step-by-step visual debugging.

**Architecture:** Python design code emits a sequence of typed IR operations (primitives, booleans, transforms, fillets, queries, exports). An OCCT backend executes the IR against build123d. A tag system (inspired by Zoo/KCL) lets edges be named at creation time and referenced later for fillets/chamfers. Each body's IR sequence is content-hashable for caching. A Rerun-based debugger visualizes execution step-by-step.

**Tech Stack:** Python 3.11+, build123d (OCCT), pytest, Rerun SDK, existing bot designs as proof points.

**Review fixes applied:** (1) Backend uses `.moved()` not `.locate()` per memory/feedback_build123d_locate.md. (2) Consistent `_ensure_solid` — reuse from `botcad.cad_utils` rather than re-implementing. (3) `_orient_z_to_axis` faithfully reproduced — orientation applied unconditionally after cylinder branch, matching cad.py:770-771. (4) `_bboxes_overlap` guard preserved in IR emitter for clearance/wire cuts. (5) `CadProgram.prebuilt_solids` is a proper dataclass field, not monkey-patched. Prebuilt solid hashes are included in `content_hash()`. (6) `CadProgram.output_ref` explicitly marks the final shape. (7) Rerun tests use `pytest.importorskip`.

# botcad/ir/__init__.py
"""CadIR — intermediate representation for CAD operations."""

# botcad/ir/ops.py
"""Typed IR operations for the CAD pipeline.

Every operation is a frozen dataclass. Shape-producing ops carry a ShapeRef
that downstream ops use to reference the result. Tags are optional string
labels that propagate through boolean operations so fillets/chamfers can
reference edges by name rather than by topological index.
"""
from __future__ import annotations

from dataclasses import dataclass
from enum import Enum
from typing import Union

Vec3 = tuple[float, float, float]


class Align3(Enum):
    """3-axis alignment for primitives."""
    CENTER = "center"
    MIN_Z = "min_z"  # centered XY, min Z (bottom-aligned)


@dataclass(frozen=True)
class ShapeRef:
    """Opaque handle to a shape produced by an IR op."""
    id: str

    def __str__(self) -> str:
        return self.id


# ── Primitives ──

@dataclass(frozen=True)
class BoxOp:
    ref: ShapeRef
    width: float
    length: float
    height: float
    align: Align3 = Align3.CENTER
    tag: str | None = None


@dataclass(frozen=True)
class CylinderOp:
    ref: ShapeRef
    radius: float
    height: float
    align: Align3 = Align3.CENTER
    tag: str | None = None


@dataclass(frozen=True)
class SphereOp:
    ref: ShapeRef
    radius: float
    tag: str | None = None


# ── Booleans ──

@dataclass(frozen=True)
class FuseOp:
    ref: ShapeRef
    target: ShapeRef
    tool: ShapeRef


@dataclass(frozen=True)
class CutOp:
    ref: ShapeRef
    target: ShapeRef
    tool: ShapeRef


# ── Transforms ──

@dataclass(frozen=True)
class LocateOp:
    ref: ShapeRef
    target: ShapeRef
    pos: Vec3 = (0.0, 0.0, 0.0)
    euler_deg: Vec3 = (0.0, 0.0, 0.0)


# ── Modifications ──

@dataclass(frozen=True)
class FilletOp:
    ref: ShapeRef
    target: ShapeRef
    tags: tuple[str, ...]
    radius: float


@dataclass(frozen=True)
class ChamferOp:
    ref: ShapeRef
    target: ShapeRef
    tags: tuple[str, ...]
    size: float


# ── Queries (no ShapeRef — these return values, not shapes) ──

@dataclass(frozen=True)
class QueryVolumeOp:
    target: ShapeRef


@dataclass(frozen=True)
class QueryCentroidOp:
    target: ShapeRef


@dataclass(frozen=True)
class QueryInertiaOp:
    target: ShapeRef


@dataclass(frozen=True)
class QueryBBoxOp:
    target: ShapeRef


@dataclass(frozen=True)
class QueryAreaOp:
    target: ShapeRef


# ── Export ──

@dataclass(frozen=True)
class ExportSTLOp:
    target: ShapeRef
    path: str


@dataclass(frozen=True)
class ExportSTEPOp:
    targets: tuple[ShapeRef, ...]
    path: str


# Union of all ops for type narrowing
PrimitiveOp = Union[BoxOp, CylinderOp, SphereOp]
BooleanOp = Union[FuseOp, CutOp]
TransformOp = LocateOp
ModificationOp = Union[FilletOp, ChamferOp]
QueryOp = Union[QueryVolumeOp, QueryCentroidOp, QueryInertiaOp, QueryBBoxOp, QueryAreaOp]
ExportOp = Union[ExportSTLOp, ExportSTEPOp]
CadOp = Union[PrimitiveOp, BooleanOp, TransformOp, ModificationOp, QueryOp, ExportOp]
```

# botcad/ir/program.py
"""CadProgram — ordered op sequence with builder API and content hashing."""
from __future__ import annotations

import hashlib
import json
from dataclasses import dataclass, field

from botcad.ir.ops import (
    Align3,
    BoxOp,
    CadOp,
    ChamferOp,
    CutOp,
    CylinderOp,
    ExportSTEPOp,
    ExportSTLOp,
    FilletOp,
    FuseOp,
    LocateOp,
    QueryAreaOp,
    QueryBBoxOp,
    QueryCentroidOp,
    QueryInertiaOp,
    QueryVolumeOp,
    ShapeRef,
    SphereOp,
    Vec3,
)


@dataclass
class CadProgram:
    """Ordered sequence of IR ops with a builder API.

    Usage:
        prog = CadProgram()
        box = prog.box(0.06, 0.04, 0.02)
        hole = prog.cylinder(0.01, 0.1, tag="pocket")
        hole = prog.locate(hole, pos=(0.01, 0, 0))
        result = prog.cut(box, hole)
        prog.query_volume(result)
        prog.output_ref = result
    """

    ops: list[CadOp] = field(default_factory=list)
    output_ref: ShapeRef | None = None  # explicit final shape
    prebuilt_solids: dict[str, object] = field(default_factory=dict)  # ref.id → solid
    _counter: int = field(default=0, repr=False)

    def _next_ref(self, prefix: str) -> ShapeRef:
        ref = ShapeRef(f"{prefix}_{self._counter}")
        self._counter += 1
        return ref

    # ── Primitives ──

    def box(
        self, width: float, length: float, height: float,
        align: Align3 = Align3.CENTER, tag: str | None = None,
    ) -> ShapeRef:
        ref = self._next_ref("box")
        self.ops.append(BoxOp(ref=ref, width=width, length=length, height=height, align=align, tag=tag))
        return ref

    def cylinder(
        self, radius: float, height: float,
        align: Align3 = Align3.CENTER, tag: str | None = None,
    ) -> ShapeRef:
        ref = self._next_ref("cyl")
        self.ops.append(CylinderOp(ref=ref, radius=radius, height=height, align=align, tag=tag))
        return ref

    def sphere(self, radius: float, tag: str | None = None) -> ShapeRef:
        ref = self._next_ref("sph")
        self.ops.append(SphereOp(ref=ref, radius=radius, tag=tag))
        return ref

    # ── Booleans ──

    def fuse(self, target: ShapeRef, tool: ShapeRef) -> ShapeRef:
        ref = self._next_ref("fuse")
        self.ops.append(FuseOp(ref=ref, target=target, tool=tool))
        return ref

    def cut(self, target: ShapeRef, tool: ShapeRef) -> ShapeRef:
        ref = self._next_ref("cut")
        self.ops.append(CutOp(ref=ref, target=target, tool=tool))
        return ref

    # ── Transforms ──

    def locate(
        self, target: ShapeRef,
        pos: Vec3 = (0.0, 0.0, 0.0), euler_deg: Vec3 = (0.0, 0.0, 0.0),
    ) -> ShapeRef:
        ref = self._next_ref("loc")
        self.ops.append(LocateOp(ref=ref, target=target, pos=pos, euler_deg=euler_deg))
        return ref

    # ── Modifications ──

    def fillet(self, target: ShapeRef, tags: tuple[str, ...], radius: float) -> ShapeRef:
        ref = self._next_ref("fillet")
        self.ops.append(FilletOp(ref=ref, target=target, tags=tags, radius=radius))
        return ref

    def chamfer(self, target: ShapeRef, tags: tuple[str, ...], size: float) -> ShapeRef:
        ref = self._next_ref("cham")
        self.ops.append(ChamferOp(ref=ref, target=target, tags=tags, size=size))
        return ref

    # ── Queries ──

    def query_volume(self, target: ShapeRef) -> None:
        self.ops.append(QueryVolumeOp(target=target))

    def query_centroid(self, target: ShapeRef) -> None:
        self.ops.append(QueryCentroidOp(target=target))

    def query_inertia(self, target: ShapeRef) -> None:
        self.ops.append(QueryInertiaOp(target=target))

    def query_bbox(self, target: ShapeRef) -> None:
        self.ops.append(QueryBBoxOp(target=target))

    def query_area(self, target: ShapeRef) -> None:
        self.ops.append(QueryAreaOp(target=target))

    # ── Export ──

    def export_stl(self, target: ShapeRef, path: str) -> None:
        self.ops.append(ExportSTLOp(target=target, path=path))

    def export_step(self, targets: tuple[ShapeRef, ...], path: str) -> None:
        self.ops.append(ExportSTEPOp(targets=targets, path=path))

    # ── Hashing & Serialization ──

    def content_hash(self) -> str:
        """SHA-256 of the canonical JSON representation.

        Includes prebuilt solid hashes so cache invalidates when
        bracket/component geometry changes.
        """
        # Include prebuilt solid hashes in the content hash
        prebuilt_hashes = {k: getattr(op, 'solid_hash', '')
                          for op in self.ops
                          for k in ([op.ref.id] if hasattr(op, 'solid_hash') else [])}
        canonical = self.to_json() + json.dumps(prebuilt_hashes, sort_keys=True)
        return hashlib.sha256(canonical.encode()).hexdigest()

    def to_json(self) -> str:
        """Canonical JSON serialization (deterministic key order)."""
        from dataclasses import asdict
        ops_data = []
        for op in self.ops:
            d = asdict(op)
            d["_type"] = type(op).__name__
            ops_data.append(d)
        return json.dumps(ops_data, sort_keys=True, separators=(",", ":"))

    @classmethod
    def from_json(cls, json_str: str) -> CadProgram:
        """Deserialize from canonical JSON."""
        import botcad.ir.ops as ops_mod

        ops_data = json.loads(json_str)
        prog = cls()
        max_counter = 0
        for d in ops_data:
            type_name = d.pop("_type")
            op_cls = getattr(ops_mod, type_name)
            # Reconstruct ShapeRef fields
            for key, val in list(d.items()):
                if isinstance(val, dict) and "id" in val:
                    d[key] = ShapeRef(val["id"])
                elif isinstance(val, list) and val and isinstance(val[0], dict):
                    d[key] = tuple(ShapeRef(v["id"]) for v in val)
                elif isinstance(val, list):
                    d[key] = tuple(val)
            # Reconstruct enums
            if "align" in d and isinstance(d["align"], str):
                d["align"] = Align3(d["align"])
            op = op_cls(**d)
            prog.ops.append(op)
            if hasattr(op, "ref"):
                # Track counter from ref IDs
                parts = op.ref.id.rsplit("_", 1)
                if len(parts) == 2 and parts[1].isdigit():
                    max_counter = max(max_counter, int(parts[1]) + 1)
        prog._counter = max_counter
        return prog
```

# botcad/ir/tags.py
"""Tag registry for tracking named edges through IR operations.

Tags are declared on primitive ops (e.g. a cylinder tagged "pocket").
When that shape participates in boolean ops, the resulting shape inherits
the tag. The OCCT backend uses tags to resolve edge selections for
fillet/chamfer operations.
"""
from __future__ import annotations

from dataclasses import dataclass, field

from botcad.ir.ops import ShapeRef


@dataclass
class TagRegistry:
    """Tracks tag declarations and propagation through operations."""

    # tag_name → source ShapeRef (the primitive that declared it)
    _declarations: dict[str, ShapeRef] = field(default_factory=dict)
    # ShapeRef → set of tag names present on that shape
    _shape_tags: dict[ShapeRef, set[str]] = field(default_factory=dict)

    def declare(self, tag: str, ref: ShapeRef) -> None:
        """Declare a new tag on a primitive shape."""
        self._declarations[tag] = ref
        self._shape_tags.setdefault(ref, set()).add(tag)

    def source_ref(self, tag: str) -> ShapeRef:
        """Get the original ShapeRef that declared this tag."""
        if tag not in self._declarations:
            raise KeyError(f"Unknown tag: {tag!r}")
        return self._declarations[tag]

    def tags_on(self, ref: ShapeRef) -> frozenset[str]:
        """Get all tags present on a shape (including propagated)."""
        return frozenset(self._shape_tags.get(ref, set()))

    def propagate_boolean(
        self, result_ref: ShapeRef, target_ref: ShapeRef, tool_ref: ShapeRef,
    ) -> None:
        """Result of boolean op inherits tags from both operands."""
        inherited = set()
        inherited.update(self._shape_tags.get(target_ref, set()))
        inherited.update(self._shape_tags.get(tool_ref, set()))
        if inherited:
            self._shape_tags.setdefault(result_ref, set()).update(inherited)

    def propagate_transform(self, result_ref: ShapeRef, source_ref: ShapeRef) -> None:
        """Result of transform inherits tags from source."""
        source_tags = self._shape_tags.get(source_ref, set())
        if source_tags:
            self._shape_tags.setdefault(result_ref, set()).update(source_tags)

# botcad/ir/backend_occt.py
"""OCCT backend — executes CadIR programs against build123d.

Walks the op list sequentially, maintaining a shape table (ShapeRef → Solid)
and a tag registry. Returns an ExecutionResult with query answers and
the final shape table.
"""
from __future__ import annotations

import logging
from dataclasses import dataclass, field
from typing import Any

from botcad.ir.ops import (
    Align3,
    BoxOp,
    ChamferOp,
    CutOp,
    CylinderOp,
    ExportSTEPOp,
    ExportSTLOp,
    FilletOp,
    FuseOp,
    LocateOp,
    QueryAreaOp,
    QueryBBoxOp,
    QueryCentroidOp,
    QueryInertiaOp,
    QueryVolumeOp,
    ShapeRef,
    SphereOp,
)
from botcad.ir.program import CadProgram
from botcad.ir.tags import TagRegistry

log = logging.getLogger(__name__)


@dataclass
class ExecutionResult:
    """Result of executing a CadProgram."""

    shapes: dict[str, object] = field(default_factory=dict)  # ref.id → build123d Solid
    queries: list[Any] = field(default_factory=list)  # ordered query results
    tags: TagRegistry = field(default_factory=TagRegistry)


    # Reuse existing battle-tested helpers from botcad.emit.cad
    # _ensure_solid, _to_compound, _bool_cut, _bboxes_overlap
    # Do NOT re-implement — import them.


class OcctBackend:
    """Executes CadProgram ops against build123d/OCCT."""

    def execute(self, program: CadProgram) -> ExecutionResult:
        from build123d import (
            Align,
            Axis,
            Box,
            Compound,
            Cylinder,
            Location,
            Sphere,
            Unit,
            export_stl,
            export_step,
        )

        result = ExecutionResult()
        shapes = result.shapes
        tags = result.tags

        def _align(a: Align3):
            if a == Align3.CENTER:
                return (Align.CENTER, Align.CENTER, Align.CENTER)
            elif a == Align3.MIN_Z:
                return (Align.CENTER, Align.CENTER, Align.MIN)

        for op in program.ops:
            match op:
                # ── Primitives ──
                case BoxOp(ref=ref, width=w, length=l, height=h, align=align, tag=tag):
                    s = Box(w, l, h, align=_align(align))
                    shapes[ref.id] = s
                    if tag:
                        tags.declare(tag, ref)

                case CylinderOp(ref=ref, radius=r, height=h, align=align, tag=tag):
                    s = Cylinder(r, h, align=_align(align))
                    shapes[ref.id] = s
                    if tag:
                        tags.declare(tag, ref)

                case SphereOp(ref=ref, radius=r, tag=tag):
                    s = Sphere(r)
                    shapes[ref.id] = s
                    if tag:
                        tags.declare(tag, ref)

                # ── Booleans ──
                case FuseOp(ref=ref, target=t, tool=tl):
                    from botcad.cad_utils import as_solid
                    from botcad.emit.cad import _ensure_solid
                    a = shapes[t.id]
                    b = shapes[tl.id]
                    s = as_solid(a.fuse(b))
                    shapes[ref.id] = s
                    tags.propagate_boolean(ref, t, tl)

                case CutOp(ref=ref, target=t, tool=tl):
                    from botcad.emit.cad import _ensure_solid, _bboxes_overlap
                    a = shapes[t.id]
                    b = shapes[tl.id]
                    # Guard: skip cut if bounding boxes don't overlap (prevents OCCT hangs)
                    if _bboxes_overlap(a, b):
                        s = _ensure_solid(a - b)
                    else:
                        s = a
                    shapes[ref.id] = s
                    tags.propagate_boolean(ref, t, tl)

                # ── Transforms ──
                case LocateOp(ref=ref, target=t, pos=pos, euler_deg=euler):
                    s = shapes[t.id]
                    loc = Location(pos, euler) if any(e != 0 for e in euler) else Location(pos)
                    # CRITICAL: use .moved() not .locate() — .locate() mutates in place
                    # and corrupts shapes referenced by multiple ops or cached shapes.
                    # See memory/feedback_build123d_locate.md.
                    shapes[ref.id] = s.moved(loc)
                    tags.propagate_transform(ref, t)

                # ── Modifications ──
                case FilletOp(ref=ref, target=t, tags=ftags, radius=radius):
                    s = shapes[t.id]
                    # Resolve tagged edges: find edges generated by the tagged shape
                    edges = _resolve_tagged_edges(s, ftags, tags, shapes)
                    if edges:
                        shapes[ref.id] = s.fillet(radius, edges)
                    else:
                        log.warning("Fillet: no edges resolved for tags %s, passing through", ftags)
                        shapes[ref.id] = s
                    # Fillet result inherits all tags from source
                    tags.propagate_transform(ref, t)

                case ChamferOp(ref=ref, target=t, tags=ctags, size=size):
                    s = shapes[t.id]
                    edges = _resolve_tagged_edges(s, ctags, tags, shapes)
                    if edges:
                        shapes[ref.id] = s.chamfer(edges, size)
                    else:
                        log.warning("Chamfer: no edges resolved for tags %s, passing through", ctags)
                        shapes[ref.id] = s
                    tags.propagate_transform(ref, t)

                # ── Queries ──
                case QueryVolumeOp(target=t):
                    s = shapes[t.id]
                    result.queries.append(abs(s.volume))

                case QueryCentroidOp(target=t):
                    s = shapes[t.id]
                    c = s.center()
                    result.queries.append((c.X, c.Y, c.Z))

                case QueryInertiaOp(target=t):
                    s = shapes[t.id]
                    m = s.matrix_of_inertia
                    result.queries.append([[m[i][j] for j in range(3)] for i in range(3)])

                case QueryBBoxOp(target=t):
                    s = shapes[t.id]
                    bb = s.bounding_box()
                    result.queries.append(
                        ((bb.min.X, bb.min.Y, bb.min.Z), (bb.max.X, bb.max.Y, bb.max.Z))
                    )

                case QueryAreaOp(target=t):
                    s = shapes[t.id]
                    result.queries.append(abs(s.area))

                # ── Export ──
                case ExportSTLOp(target=t, path=path):
                    export_stl(shapes[t.id], path, unit=Unit.METER)

                case ExportSTEPOp(targets=ts, path=path):
                    solids = [shapes[t.id] for t in ts]
                    compound = Compound(children=solids)
                    export_step(compound, path)

                case _:
                    raise ValueError(f"Unknown op: {op}")

        return result


def _resolve_tagged_edges(solid, tag_names, tag_registry, shapes):
    """Resolve tag names to actual OCCT edges on the solid.

    Strategy: For each tag, get the source shape (the primitive that declared
    the tag). Use its bounding box to filter edges on the target solid that
    are geometrically close to the tagged shape.

    This is a heuristic — OCCT doesn't natively track "which edges came from
    which boolean operand." We use proximity to the tool shape's geometry
    as the selection criterion.
    """
    from build123d import Axis

    all_edges = []
    solid_edges = solid.edges()

    for tag_name in tag_names:
        source_ref = tag_registry.source_ref(tag_name)
        source_shape = shapes.get(source_ref.id)

        if source_shape is None:
            continue

        # For primitives used as boolean tools: edges near the tool surface
        # are the ones created by the boolean op. Use the tool's bounding box
        # as a spatial filter.
        try:
            tool_bb = source_shape.bounding_box()
        except Exception:
            # Fallback: return all edges (will over-fillet but won't crash)
            all_edges.extend(solid_edges)
            continue

        # Expand tool bbox slightly and filter solid edges by containment
        margin = 0.001  # 1mm tolerance
        for edge in solid_edges:
            try:
                ec = edge.center()
                if (
                    tool_bb.min.X - margin <= ec.X <= tool_bb.max.X + margin
                    and tool_bb.min.Y - margin <= ec.Y <= tool_bb.max.Y + margin
                    and tool_bb.min.Z - margin <= ec.Z <= tool_bb.max.Z + margin
                ):
                    all_edges.append(edge)
            except Exception:
                continue

    return all_edges if all_edges else None

# botcad/ir/cache.py
"""Disk cache for CadIR execution results.

Keyed by CadProgram.content_hash(). Stores arbitrary picklable data
(typically mass properties + STL mesh bytes).
"""
from __future__ import annotations

import logging
import pickle
import shutil
from pathlib import Path
from typing import Any

from botcad.ir.program import CadProgram

log = logging.getLogger(__name__)


class CadCache:
    """Simple disk cache keyed by program content hash."""

    def __init__(self, cache_dir: Path) -> None:
        self._dir = Path(cache_dir)
        self._dir.mkdir(parents=True, exist_ok=True)

    def _path(self, prog: CadProgram) -> Path:
        h = prog.content_hash()
        return self._dir / f"{h}.pkl"

    def get(self, prog: CadProgram) -> dict[str, Any] | None:
        p = self._path(prog)
        if not p.exists():
            return None
        try:
            return pickle.loads(p.read_bytes())
        except Exception:
            log.warning("Cache read failed for %s, treating as miss", p.name)
            return None

    def put(self, prog: CadProgram, data: dict[str, Any]) -> None:
        p = self._path(prog)
        p.write_bytes(pickle.dumps(data))

    def invalidate(self, prog: CadProgram) -> None:
        p = self._path(prog)
        p.unlink(missing_ok=True)

    def clear(self) -> None:
        if self._dir.exists():
            shutil.rmtree(self._dir)
            self._dir.mkdir(parents=True, exist_ok=True)
```

def _total_volume(solid) -> float:
    solids = solid.solids() if hasattr(solid, "solids") else [solid]
    return sum(abs(s.volume) for s in solids)

# botcad/ir/emit_body.py
"""Emit CadIR for a single body solid.

Translates the logic from _make_body_solid() (botcad/emit/cad.py:730-883)
into a CadProgram. Bracket, coupler, and component solids are injected
as PrebuiltOp since they have their own build123d factories.
"""
from __future__ import annotations

import hashlib
from typing import TYPE_CHECKING

from botcad.ir.ops import Align3, PrebuiltOp, ShapeRef
from botcad.ir.program import CadProgram

if TYPE_CHECKING:
    from botcad.skeleton import Body, Joint


def emit_body_ir(
    body: Body,
    parent_joint: Joint | None = None,
    wire_segments: tuple | None = None,
) -> CadProgram:
    """Emit a CadProgram that builds the body solid.

    Returns a CadProgram whose final shape-producing op is the complete
    body solid (same geometry as _make_body_solid would produce).
    """
    from botcad.bracket import (
        BracketSpec,
        bracket_envelope,
        bracket_solid,
        coupler_solid,
        cradle_envelope,
        cradle_solid,
    )
    from botcad.component import BearingSpec, BusType, CameraSpec
    from botcad.geometry import quat_to_euler, rotate_vec
    from botcad.skeleton import BodyShape, BracketStyle

    prog = CadProgram()
    bracket_spec = BracketSpec()
    dims = body.dimensions

    # ── 1. Base shell ──
    shell = _emit_base_shell(prog, body, parent_joint, dims)

    # ── 2. Component pockets ──
    if not body.is_wheel_body:
        for mount in body.mounts:
            cd = mount.placed_dimensions
            if isinstance(mount.component, BearingSpec):
                tol = 0.0005
                pocket_r = mount.component.od / 2 + tol / 2
                pocket_h = mount.component.width + 0.001
                pocket = prog.cylinder(pocket_r, pocket_h, tag=f"pocket_{mount.label}")
            else:
                pocket = prog.box(
                    cd[0] + 0.0005, cd[1] + 0.0005, cd[2] + 0.0005,
                    tag=f"pocket_{mount.label}",
                )
            pocket = prog.locate(pocket, pos=mount.resolved_pos)
            shell = prog.cut(shell, pocket)

            if isinstance(mount.component, CameraSpec):
                shell = _emit_camera_cuts(prog, shell, mount, body.dimensions, dims)

    # ── 3. Coupler union (child body with coupler-style parent) ──
    if parent_joint is not None and parent_joint.bracket_style is BracketStyle.COUPLER:
        servo = parent_joint.servo
        quat = parent_joint.solved_servo_quat
        rotated_offset = rotate_vec(quat, servo.shaft_offset)
        center = (-rotated_offset[0], -rotated_offset[1], -rotated_offset[2])
        euler = quat_to_euler(quat)
        coupler = coupler_solid(servo, bracket_spec)
        coupler_ref = _inject_prebuilt(prog, coupler, tag="coupler")
        coupler_ref = prog.locate(coupler_ref, pos=center, euler_deg=euler)
        shell = prog.fuse(shell, coupler_ref)

    # ── 4. Bracket footprint cut + bracket union at each joint ──
    for joint in body.joints:
        servo = joint.servo
        center = joint.solved_servo_center
        euler = quat_to_euler(joint.solved_servo_quat)

        if joint.bracket_style is BracketStyle.COUPLER:
            env = cradle_envelope(servo, bracket_spec)
            env_ref = _inject_prebuilt(prog, env, tag=f"cradle_env_{joint.name}")
            env_ref = prog.locate(env_ref, pos=center, euler_deg=euler)
            cradle = cradle_solid(servo, bracket_spec)
            cradle_ref = _inject_prebuilt(prog, cradle, tag=f"cradle_{joint.name}")
            cradle_ref = prog.locate(cradle_ref, pos=center, euler_deg=euler)
            cut_result = prog.cut(shell, env_ref)
            shell = prog.fuse(cut_result, cradle_ref)
        else:
            env = bracket_envelope(servo, bracket_spec)
            env_ref = _inject_prebuilt(prog, env, tag=f"bracket_env_{joint.name}")
            env_ref = prog.locate(env_ref, pos=center, euler_deg=euler)
            bracket = bracket_solid(servo, bracket_spec)
            brk_ref = _inject_prebuilt(prog, bracket, tag=f"bracket_{joint.name}")
            brk_ref = prog.locate(brk_ref, pos=center, euler_deg=euler)
            cut_result = prog.cut(shell, env_ref)
            shell = prog.fuse(cut_result, brk_ref)

    # ── 5. Child clearance volumes ──
    # NOTE: _bboxes_overlap guard is applied at execution time by the backend
    # (CutOp handler checks overlap before calling OCCT). See backend_occt.py.
    for joint in body.joints:
        if joint.child is not None:
            from botcad.emit.cad import _child_clearance_volume
            clearance = _child_clearance_volume(joint.child, joint)
            if clearance is not None:
                clr_ref = _inject_prebuilt(prog, clearance, tag=f"clearance_{joint.name}")
                shell = prog.cut(shell, clr_ref)

    # ── 6. Wire channels ──
    # Same: _bboxes_overlap guard is in the backend CutOp handler.
    if wire_segments:
        for seg, bus_type in wire_segments:
            from botcad.emit.cad import _wire_channel
            channel = _wire_channel(seg, bus_type)
            if channel is not None:
                ch_ref = _inject_prebuilt(prog, channel, tag=f"wire_{seg.body_name}")
                shell = prog.cut(shell, ch_ref)

    prog.output_ref = shell
    return prog


def _emit_base_shell(prog, body, parent_joint, dims):
    """Emit IR ops for the base shell shape."""
    from botcad.skeleton import BodyShape

    if body.custom_solid is not None:
        return _inject_prebuilt(prog, body.custom_solid)

    if body.shape is BodyShape.CYLINDER:
        r = body.radius or dims[0] / 2
        h = body.width or dims[2]
        if body.is_wheel_body:
            from botcad.emit.cad import _make_wheel_solid
            wheel = _make_wheel_solid(r, h)
            ref = _inject_prebuilt(prog, wheel, tag="wheel")
        elif parent_joint is not None:
            ref = prog.cylinder(r, h)
            ref = prog.locate(ref, pos=(0, 0, h / 2))
        else:
            ref = prog.cylinder(r, h)

        # CRITICAL: orient Z to joint axis UNCONDITIONALLY after all cylinder
        # sub-branches. Matches cad.py:770-771 which runs _orient_z_to_axis
        # outside the if/elif/else block for all cylinder bodies with a parent.
        if parent_joint is not None:
            euler = _axis_to_euler(parent_joint.axis, body.frame_quat)
            if any(e != 0 for e in euler):
                ref = prog.locate(ref, euler_deg=euler)
        return ref

    elif body.shape is BodyShape.TUBE:
        r = body.outer_r or dims[0] / 2
        length = body.length or dims[2]
        ref = prog.cylinder(r, length)
        ref = prog.locate(ref, pos=(0, 0, length / 2))
        return ref

    elif body.shape is BodyShape.SPHERE:
        r = body.radius or dims[0] / 2
        return prog.sphere(r)

    elif body.shape is BodyShape.JAW:
        jw = body.jaw_width or dims[0]
        jt = body.jaw_thickness or dims[1]
        jl = body.jaw_length or dims[2]
        knuckle_h = 0.008
        knuckle = prog.box(jw, jt * 2, knuckle_h, align=Align3.MIN_Z)
        plate = prog.box(jw, jt, jl - knuckle_h, align=Align3.MIN_Z)
        plate = prog.locate(plate, pos=(0, 0, knuckle_h))
        return prog.fuse(knuckle, plate)

    else:  # BOX
        return prog.box(dims[0], dims[1], dims[2])


def _emit_camera_cuts(prog, shell, mount, body_dims, solved_dims):
    """Emit IR ops for camera lens aperture and ribbon slot cuts."""
    from botcad.component import BusType

    cd = mount.component.dimensions
    pos = mount.resolved_pos
    ins = mount.resolved_insertion_axis

    aperture_r = 0.004
    wall_depth = max(solved_dims) + 0.002

    ax, ay, az = ins
    if abs(ax) > 0.5:
        euler = (0.0, 90.0, 0.0)
    elif abs(ay) > 0.5:
        euler = (90.0, 0.0, 0.0)
    else:
        euler = (0.0, 0.0, 0.0)

    aperture = prog.cylinder(aperture_r, wall_depth, tag="lens_aperture")
    if any(e != 0 for e in euler):
        aperture = prog.locate(aperture, euler_deg=euler)
    aperture = prog.locate(aperture, pos=pos)
    shell = prog.cut(shell, aperture)

    ribbon_w = 0.017
    ribbon_h = 0.002
    csi_ports = [wp for wp in mount.component.wire_ports if wp.bus_type == BusType.CSI]
    if csi_ports:
        wp = csi_ports[0]
        slot_pos = (pos[0] + wp.pos[0], pos[1] + wp.pos[1], pos[2] + wp.pos[2])
    else:
        slot_pos = (pos[0], pos[1], pos[2] - cd[1] / 2)

    if abs(ax) > 0.5:
        slot = prog.box(wall_depth, ribbon_w, ribbon_h, tag="ribbon_slot")
    elif abs(ay) > 0.5:
        slot = prog.box(ribbon_w, wall_depth, ribbon_h, tag="ribbon_slot")
    else:
        slot = prog.box(ribbon_w, ribbon_h, wall_depth, tag="ribbon_slot")

    slot = prog.locate(slot, pos=slot_pos)
    shell = prog.cut(shell, slot)

    return shell


def _inject_prebuilt(prog, solid, tag=None):
    """Inject a pre-built solid into the program as a PrebuiltOp."""
    # Hash actual volume + bbox for reliable cache invalidation
    try:
        bb = solid.bounding_box()
        vol = solid.volume
        hash_input = f"{vol:.10f}|{bb.min.X:.6f},{bb.min.Y:.6f},{bb.min.Z:.6f}|{bb.max.X:.6f},{bb.max.Y:.6f},{bb.max.Z:.6f}"
    except Exception:
        hash_input = repr(id(solid))
    solid_hash = hashlib.sha256(hash_input.encode()).hexdigest()[:16]
    ref = prog._next_ref("pre")
    prog.ops.append(PrebuiltOp(ref=ref, solid_hash=solid_hash, tag=tag))
    # Store via proper dataclass field (not monkey-patching)
    prog.prebuilt_solids[ref.id] = solid
    return ref


def _axis_to_euler(axis, frame_quat):
    """Convert axis + frame_quat to Euler angles for IR locate op."""
    from botcad.geometry import quat_to_euler
    w, x, y, z = frame_quat
    if abs(w - 1.0) < 1e-9 and abs(x) + abs(y) + abs(z) < 1e-9:
        return (0.0, 0.0, 0.0)
    return quat_to_euler(frame_quat)

"""Rerun-based step-by-step debugger for CadIR programs.

Each IR op maps to a Rerun timestep. Shape-producing ops export their
mesh and log it as a 3D entity. Boolean ops log both operands (tool in
red overlay) plus the result.
"""
from __future__ import annotations

import logging
from pathlib import Path
from typing import TYPE_CHECKING

from botcad.ir.backend_occt import OcctBackend, ExecutionResult
from botcad.ir.ops import (
    BoxOp, CutOp, CylinderOp, FilletOp, FuseOp,
    LocateOp, PrebuiltOp, SphereOp, ChamferOp,
)
from botcad.ir.program import CadProgram

if TYPE_CHECKING:
    pass

log = logging.getLogger(__name__)


def debug_program(
    prog: CadProgram,
    spawn_viewer: bool = True,
    mesh_dir: Path | None = None,
) -> None:
    """Execute program step-by-step, logging each op to Rerun.

    Args:
        prog: The CadProgram to debug.
        spawn_viewer: If True, launch the Rerun viewer.
        mesh_dir: If set, export per-step STL files here.
    """
    import rerun as rr
    from build123d import export_stl, Unit

    rr.init("botcad_ir_debug", spawn=spawn_viewer)
    if not spawn_viewer:
        rr.set_global_data_recording(rr.new_recording(application_id="botcad_ir_debug"))

    # Execute the full program first to get all shapes
    backend = OcctBackend()
    result = backend.execute(prog)

    if mesh_dir is not None:
        mesh_dir = Path(mesh_dir)
        mesh_dir.mkdir(parents=True, exist_ok=True)

    for step, op in enumerate(prog.ops):
        rr.set_time("step", sequence=step)

        # Log op metadata
        op_desc = f"{type(op).__name__}"
        if hasattr(op, "ref"):
            op_desc += f" → {op.ref.id}"
        rr.log("ops/description", rr.TextLog(op_desc))

        # Log meshes for shape-producing ops
        if hasattr(op, "ref") and op.ref.id in result.shapes:
            solid = result.shapes[op.ref.id]
            try:
                if mesh_dir is not None:
                    stl_path = mesh_dir / f"step_{step:03d}_{op.ref.id}.stl"
                    export_stl(solid, str(stl_path), unit=Unit.METER)

                # Log as Rerun mesh
                _log_solid_to_rerun(rr, solid, f"shapes/{op.ref.id}", step)

                # For boolean ops, also log the operands in a different color
                if isinstance(op, (CutOp, FuseOp)):
                    if op.tool.id in result.shapes:
                        _log_solid_to_rerun(
                            rr, result.shapes[op.tool.id],
                            f"shapes/{op.tool.id}_tool", step,
                        )
            except Exception as e:
                log.warning("Failed to log step %d: %s", step, e)


def _log_solid_to_rerun(rr, solid, entity_path, step):
    """Convert a build123d Solid to Rerun mesh and log it."""
    try:
        tess = solid.tessellate(0.001)
        if tess and len(tess) == 2:
            vertices, faces = tess
            rr.log(
                entity_path,
                rr.Mesh3D(
                    vertex_positions=vertices,
                    triangle_indices=faces,
                ),
            )
    except Exception as e:
        log.debug("Tessellation failed for %s: %s", entity_path, e)
```
